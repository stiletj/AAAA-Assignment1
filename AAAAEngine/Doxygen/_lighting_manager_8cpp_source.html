152void LightingManager::SetPointUniforms(Shader& shader, glm::mat4 worldMatrix)
153{
154 m_pointLightColours.clear();
155 std::vector<glm::vec3> positions;
156 for (int i = 0; i < m_maxLights; i++)

160 positions.push_back(m_pointLights[i].GetLocalPosition(worldMatrix));

164 }
165
166 shader.SetUniform3fv("u_PointPositions", m_maxLights, positions);
167 shader.SetUniform4fv("u_PointColours", m_maxLights, m_pointLightColours);
168 shader.SetUniform1fv("u_PointAmbientIntensities", m_maxLights, m_ambientIntensies.data());
169 shader.SetUniform1fv("u_PointDiffuseIntensities", m_maxLights, m_diffuseIntensies.data());
170 shader.SetUniform1fv("u_PointConstants", m_maxLights, m_constants.data());
171 shader.SetUniform1fv("u_PointLinears", m_maxLights, m_linears.data());
172 shader.SetUniform1fv("u_PointExps", m_maxLights, m_exps.data());
173 shader.SetUniform1i("u_NumOfPoints", m_activePointLights.size());
