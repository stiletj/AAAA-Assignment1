40// Function to handle keyboard input for player movement and camera control
41void processInput(GLFWwindow* window) {
42 // Forward/backward movement
43 if (glfwGetKey(window, GLFW_KEY_W) == GLFW_PRESS)
44 cameraPosition += cameraSpeed * cameraFront;
45 if (glfwGetKey(window, GLFW_KEY_S) == GLFW_PRESS)
46 cameraPosition -= cameraSpeed * cameraFront;
47
48 // Strafe movement
49 if (glfwGetKey(window, GLFW_KEY_A) == GLFW_PRESS)
50 cameraPosition -= glm::normalize(glm::cross(cameraFront, cameraUp)) * cameraSpeed;
51 if (glfwGetKey(window, GLFW_KEY_D) == GLFW_PRESS)
52 cameraPosition += glm::normalize(glm::cross(cameraFront, cameraUp)) * cameraSpeed;
53
54 // Camera rotation
55 const float sensitivity = 0.1f;
56 if (glfwGetKey(window, GLFW_KEY_LEFT) == GLFW_PRESS)
57 yaw -= sensitivity;
58 if (glfwGetKey(window, GLFW_KEY_RIGHT) == GLFW_PRESS)
59 yaw += sensitivity;
60 if (glfwGetKey(window, GLFW_KEY_UP) == GLFW_PRESS)
61 pitch += sensitivity;
62 if (glfwGetKey(window, GLFW_KEY_DOWN) == GLFW_PRESS)
63 pitch -= sensitivity;
64
65 // Clamp pitch to prevent camera flipping
66 if (pitch > 89.0f)
67 pitch = 89.0f;
68 if (pitch < -89.0f)
69 pitch = -89.0f;
70
71 // Update camera front vector based on yaw and pitch
72 glm::vec3 front;
73 front.x = cos(glm::radians(yaw)) * cos(glm::radians(pitch));
74 front.y = sin(glm::radians(pitch));
75 front.z = sin(glm::radians(yaw)) * cos(glm::radians(pitch));
76 cameraFront = glm::normalize(front);
77
78 // Toggle between solid color and wireframe views when 'c' key is pressed
79 static bool cKeyPressed = false;
80 if (glfwGetKey(window, GLFW_KEY_C) == GLFW_PRESS) {
81 if (!cKeyPressed) {
82 if (renderingMode) {
83 // Switch to wireframe
84 glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
85 renderingMode = 0;
86 }
87 else {
88 // Switch to solid color
89 glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
90 renderingMode = 1;
91 }
92 cKeyPressed = true;
93 }
94 }
95 else {
96 cKeyPressed = false;
97 }
98}
